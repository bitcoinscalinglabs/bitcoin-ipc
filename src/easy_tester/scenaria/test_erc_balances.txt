# Test ERC20 balance tracking, supply adjustments, and firewall enforcement
#
# Scenario tests:
# 1. Registration seeds S_home balance with initialSupply
# 2. Transfer debits source, credits destination
# 3. Firewall: transfer exceeding balance is dropped
# 4. Mint increases S_home balance
# 5. Burn decreases S_home balance
# 6. Multi-token isolation (token A balance doesn't affect token B)

config
tester ErcTransferTester

setup
validator1
validator2
validator3
validator4
subnet1
whitelist validator1 validator2 validator3 validator4
min 4
subnet2
whitelist validator1 validator2 validator3 validator4
min 4

scenario

block 1
create subnet1
create subnet2

block 2
join subnet1 validator1 100_000_000
join subnet1 validator2 100_000_000
join subnet1 validator3 100_000_000
join subnet1 validator4 100_000_000

block 3
join subnet2 validator1 100_000_000
join subnet2 validator2 100_000_000
join subnet2 validator3 100_000_000
join subnet2 validator4 100_000_000

# ──────────────────────────────────────────────────
# Test 1: Registration seeds balance
# ──────────────────────────────────────────────────

block 10
register_token subnet1 TokenA TKA 18
checkpoint subnet1

# subnet1 (home) should have balance = initialSupply (1,000,000)
read token_balance subnet1 TokenA
expect result.balance = 1000000

# subnet2 should have balance = 0
read token_balance subnet2 TokenA
expect result.balance = 0

# ──────────────────────────────────────────────────
# Test 2: Valid transfer debits/credits
# ──────────────────────────────────────────────────

block 11
erc_transfer subnet1 subnet2 TokenA 100000
checkpoint subnet1

# subnet1: 1,000,000 - 100,000 = 900,000
read token_balance subnet1 TokenA
expect result.balance = 900000

# subnet2: 0 + 100,000 = 100,000
read token_balance subnet2 TokenA
expect result.balance = 100000

# Transfer should have been delivered
read rootnet_msgs subnet2
expect result.count = 2
expect result.1.kind = erc_transfer
expect result.1.amount = 100000

# ──────────────────────────────────────────────────
# Test 3: Firewall — transfer exceeding balance dropped
# ──────────────────────────────────────────────────

block 12
erc_transfer subnet1 subnet2 TokenA 999999
checkpoint subnet1

# subnet1 balance should be unchanged (transfer was dropped)
read token_balance subnet1 TokenA
expect result.balance = 900000

# subnet2 balance unchanged
read token_balance subnet2 TokenA
expect result.balance = 100000

# No new rootnet message delivered (the invalid ETX was dropped)
read rootnet_msgs subnet2
expect result.count = 2

# ──────────────────────────────────────────────────
# Test 4: Mint increases S_home balance
# ──────────────────────────────────────────────────

block 13
mint_token subnet1 TokenA 500000
checkpoint subnet1

# subnet1: 900,000 + 500,000 = 1,400,000
read token_balance subnet1 TokenA
expect result.balance = 1400000

# Now a large transfer should succeed
block 14
erc_transfer subnet1 subnet2 TokenA 1000000
checkpoint subnet1

read token_balance subnet1 TokenA
expect result.balance = 400000

read token_balance subnet2 TokenA
expect result.balance = 1100000

# ──────────────────────────────────────────────────
# Test 5: Burn decreases S_home balance
# ──────────────────────────────────────────────────

block 15
burn_token subnet1 TokenA 300000
checkpoint subnet1

read token_balance subnet1 TokenA
expect result.balance = 100000

# ──────────────────────────────────────────────────
# Test 6: Multi-token isolation
# ──────────────────────────────────────────────────

block 16
register_token subnet1 TokenB TKB 8
checkpoint subnet1

read token_balance subnet1 TokenB
expect result.balance = 1000000

# Transfer of TokenA should not affect TokenB
block 17
erc_transfer subnet1 subnet2 TokenA 50000
checkpoint subnet1

read token_balance subnet1 TokenA
expect result.balance = 50000

# TokenB unchanged
read token_balance subnet1 TokenB
expect result.balance = 1000000

# ──────────────────────────────────────────────────
# Test 7: Multiple ETX in one checkpoint — total exceeds balance
# subnet1 has 50,000 TokenA. Two transfers of 30,000 each = 60,000 > 50,000.
# The monitor should accept the first and drop the second.
# ──────────────────────────────────────────────────

block 18
erc_transfer subnet1 subnet2 TokenA 30000
erc_transfer subnet1 subnet2 TokenA 30000
checkpoint subnet1

# First transfer goes through: 50,000 - 30,000 = 20,000
# Second transfer dropped: 20,000 < 30,000
read token_balance subnet1 TokenA
expect result.balance = 20000

# subnet2 was at 1,150,000 (1,100,000 from test 4 + 50,000 from test 6) + 30,000 = 1,180,000
read token_balance subnet2 TokenA
expect result.balance = 1180000

# ──────────────────────────────────────────────────
# Test 8: Burn exceeding balance is dropped
# subnet1 has 20,000 TokenA. Burn 50,000 should be dropped.
# ──────────────────────────────────────────────────

block 19
burn_token subnet1 TokenA 50000
checkpoint subnet1

# Balance unchanged — burn was dropped
read token_balance subnet1 TokenA
expect result.balance = 20000

# ──────────────────────────────────────────────────
# Test 9: Non-home subnet forwards tokens
# subnet2 has 1,130,000 TokenA (received from subnet1).
# subnet2 transfers 500,000 to subnet1 (back to home).
# ──────────────────────────────────────────────────

block 20
erc_transfer subnet2 subnet1 TokenA 500000
checkpoint subnet2

# subnet2: 1,180,000 - 500,000 = 680,000
read token_balance subnet2 TokenA
expect result.balance = 680000

# subnet1: 20,000 + 500,000 = 520,000
read token_balance subnet1 TokenA
expect result.balance = 520000
